    mergeSort(array: number[]) {
        if (array.length === 1) {
            return array;
        }

        const middle = Math.floor(array.length / 2);
        const leftArr = array.slice(0, middle);
        const rightArr = array.slice(middle);

        return this.merge(this.mergeSort(leftArr), this.mergeSort(rightArr));
    }
